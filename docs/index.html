Introduction

Using this dataset we aim to present an updated and meaningful report on the share and historic evolution of renewables in the European energy mix.

Our dataset comprises information for the last 20 years on the energy production and the sources which contributed to the energy mix.

data <- df %>%
  select(year = Year, country = Area, esource = Variable,
         TWh = `Generation (TWh)`, percent = `Share of production (%)`)

data %>%
  filter(country == "Romania") %>%
  filter(esource %in% c("Fossil", "Renewables", "Nuclear")) %>%
  select(!country) %>%
  group_by(year) %>%
  ggplot() +
    geom_line(aes(x = year, y = TWh, color = esource))

data %>%
  filter(esource %in% c("Fossil", "Renewables", "Nuclear")) %>%
  group_by(year, esource) %>%
  summarise(total = sum(TWh)) %>%
  ggplot() +
    geom_line(aes(x = year, y = total, color = esource))
